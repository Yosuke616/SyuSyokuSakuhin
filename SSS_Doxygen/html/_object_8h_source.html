93 template<class T>
94 T* AddComponent() {
95 T* component = new T(); //コンポーネントの生成
96 component->Parent = this; //コンポネントの持ち主の設定
97
98 //更新順序を決める
99 std::list<Component*>::iterator itr = ComponentList.begin();
100 for (auto&& com : ComponentList) {
101 //現在のオブジェクトの機能を取得
102 Component* buff = (Component*)com;
103
104 //更新の優先度が同じとき、ループが抜ける
105 if (component->m_eUpdateOrder <= buff->m_eUpdateOrder) {
106 break;
107 }
108 else {
109 //更新順序を遅らせる
110 itr++;
111 }
112 }
113
114 //コンポーネントリストに挿入
115 ComponentList.insert(itr,component);
116
117 return component;
118 }
